```typescript
// ✅ 使用QueryBuilder构建复杂查询
async findWorkOrders(query: QueryWorkOrderDTO): Promise<PaginatedResult<WorkOrder>> {
  const qb = this.repository.createQueryBuilder('wo')
    .leftJoinAndSelect('wo.material', 'm')
    .leftJoinAndSelect('wo.operations', 'op')
    .where('wo.tenantId = :tenantId', { tenantId: this.tenantContext.tenantId });

  // 动态条件
  if (query.status) {
    qb.andWhere('wo.status = :status', { status: query.status });
  }

  if (query.materialId) {
    qb.andWhere('wo.materialId = :materialId', { materialId: query.materialId });
  }

  if (query.plannedStartGte) {
    qb.andWhere('wo.plannedStart >= :start', { start: query.plannedStartGte });
  }

  // 排序
  if (query.sort) {
    const [field, order] = query.sort.startsWith('-') 
      ? [query.sort.slice(1), 'DESC'] 
      : [query.sort, 'ASC'];
    qb.orderBy(`wo.${field}`, order as 'ASC' | 'DESC');
  } else {
    qb.orderBy('wo.createdAt', 'DESC');
  }

  // 分页
  const page = query.page || 1;
  const pageSize = Math.min(query.pageSize || 20, 100);
  qb.skip((page - 1) * pageSize).take(pageSize);

  const [list, total] = await qb.getManyAndCount();

  return {
    list,
    pagination: {
      page,
      pageSize,
      total,
      totalPages: Math.ceil(total / pageSize)
    }
  };
}

// ❌ 避免原生SQL（除非性能瓶颈）
```

### 2.4 错误处理规范

```typescript
// ✅ 统一异常过滤器
@Catch()
export class GlobalExceptionFilter implements ExceptionFilter {
  catch(exception: unknown, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse();
    const request = ctx.getRequest();

    let status = HttpStatus.INTERNAL_SERVER_ERROR;
    let errorCode = 'INTERNAL_ERROR';
    let message = 'Internal server error';

    if (exception instanceof HttpException) {
      status = exception.getStatus();
      const res = exception.getResponse() as any;
      message = res.message || exception.message;
      errorCode = res.errorCode || `HTTP_${status}`;
    } else if (exception instanceof QueryFailedError) {
      status = HttpStatus.BAD_REQUEST;
      errorCode = 'DATABASE_ERROR';
      message = 'Database operation failed';
    }

    // 记录日志
    Logger.error(
      `${request.method} ${request.url}`,
      exception instanceof Error ? exception.stack : '',
      'ExceptionFilter'
    );

    response.status(status).json({
      code: status,
      errorCode,
      message,
      timestamp: new Date().toISOString(),
      requestId: request.requestId
    });
  }
}

// ✅ 业务异常
export class BusinessException extends HttpException {
  constructor(errorCode: string, message: string, status = HttpStatus.BAD_REQUEST) {
    super({ errorCode, message }, status);
  }
}

// 使用示例

  throw new BusinessException('PLM_MATERIAL_NOT_FOUND', '物料不存在', HttpStatus.NOT_FOUND);
}
```

## 6. 风险管理

### 6.1 技术风险

| 风险 | 概率 | 影响 | 应对策略 |
|:---|:---:|:---:|:---|
| 性能不达标 | 中 | 高 | 每周性能测试，慢查询优化 |
| 数据库设计缺陷 | 中 | 高 | 第2周冻结数据库设计，评审后不改 |
| 第三方库不维护 | 低 | 中 | 优先选社区活跃库，锁定版本 |
| 技术债务累积 | 高 | 中 | 每个Sprint预留20%时间重构 |

### 6.2 项目风险

| 风险 | 概率 | 影响 | 应对策略 |
|:---|:---:|:---:|:---|
| 需求变更频繁 | 高 | 高 | 冻结MVP范围，变更走评审 |
| 团队成员变动 | 低 | 高 | 代码文档齐全，知识共享 |
| 客户期望过高 | 中 | 中 | 定期演示，管理预期 |
| 进度延期 | 中 | 高 | 每周检查，提前预警，砍功能保质量 |

### 6.3 风险预警机制

```
🟢 正常：进度<100%，质量达标
🟡 预警：进度100-120%，或质量轻微下降
🔴 危险：进度>120%，或质量严重下降

预警触发动作：
- 🟡：每日站会讨论，调整任务优先级
- 🔴：紧急会议，决策砍功能或延期
```
